services:
  breadbox:
    environment:
      - TELLER_APP_ID=${TELLER_APP_ID}
      - TELLER_CERT_PATH=/certs/teller/certificate.pem
      - TELLER_KEY_PATH=/certs/teller/private_key.pem
      - TELLER_ENV=${TELLER_ENV:-sandbox}
    volumes:
      - ./certs/teller:/certs/teller:ro
